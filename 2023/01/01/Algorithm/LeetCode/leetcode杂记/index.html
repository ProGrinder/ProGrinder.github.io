判断圆形和矩形是否相交

leetcode-1401.圆和矩形是否有重叠

这题相当有意思，如果按照常理来判断，需要分多种情况讨论圆心所在位置：

  • 圆心在矩形上下左右
  • 圆形在矩形四个顶角上

但是下面这个知乎回答，使用了向量来统合了所有情况：

https://www.zhihu.com/question/24251545/answer/27184960

以下是其代码实现版：

class Solution {
public:
    bool checkOverlap(int radius, int xCenter, int yCenter, int x1, int y1, int x2, int y2) {
        double xc = (x1 + x2) / 2.0;
        double yc = (y1 + y2) / 2.0;

        vector<double> v = {abs(xCenter - xc), abs(yCenter - yc)};
        vector<double> h = {(x2 - x1) / 2.0, (y2 - y1) / 2.0};

        vector<double> u = {max(v[0] - h[0], 0.0), max(v[1] - h[1], 0.0)};

         return sqrt(u[0] * u[0] + u[1] * u[1]) <= radius;
    }
};

回环句

leetcode-2490.回环句

  • 环形写法(i + 1) % n

  • c++实现string分割函数split，类似python中自带的split

class Solution {
public:
    bool isCircularSentence(string sentence) {
        vector<string> words = split(sentence, ' ');
        int n = words.size();

        for (int i = 0; i < n; i ++) {
            if (words[i].back() != words[(i + 1) % n][0]) {
                return false;
            }
        }
        return true;
    }

   	// delimiter 分隔符
    vector<string> split(string& s, char delimiter) {
        int n = s.size();
        vector<string>words;
        int left = 0, right = 0;
        while (right < n) {
            if (s[right] == delimiter) {
                words.push_back(s.substr(left, right - left));
                left = right + 1;
            }
            right++;
        }
        words.push_back(s.substr(left, right - left));
        return words;
    }
};
