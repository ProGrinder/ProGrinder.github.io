0x60图论-(2)-最短路

源点：图中的起点

汇点：图中的终点

需要做到: 输出参数 start, 然后返回参数vector<int>返回每个点到起点的最短路径！然后选用即可。

单源最短路

单源，顾名思义就是图中只有一个起点。

单源最短路问题，需要记录额外两个变量:

int dist[N];      // 当前结点到源点的最短路径距离
bool visited[N];  // 当前结点到源点的最短路径是否已经确定

所有边权都是正数

Dijkstra算法是基于贪心的思想，适用于求所有边权都是正数的最短路径。

step1：初始化距离：起点到起点的距离为0，其它点到起点的距离为正无穷。

#include <cstring>
// 初始化距离
memset(dist, 0x3f, sizeof dist);
dist[1] = 0;

step2： 循环松弛，每次确定一个点到起点的最短路，在没有确定的点中选取一个距离起点最短的点加入到当前路径。然后用该点去更新其它未确定的点到起点的距离。

总结一下，Dijkstra算法的流程就是，不断取出离顶点最近而没有被访问过的点，松弛它和它能到达的所有点。

朴素版Dijkstra(O(n2{O(n^2}O(n2))

适用于稠密图，用邻接矩阵存储。

AcWing849.Dijkstra求最短路 I

算法模板:

#include <iostream>
#include <cstring>
#include <algorithm>

using namespace std;

const int N = 500 + 7;

int graph[N][N];
int dist[N];
bool visited[N];

int n, m;

int Dijkstra() {
    // 初始化距离
    memset(dist, 0x3f, sizeof dist);
    dist[1] = 0;
    
    for (int i = 0; i < n; i ++) {
        // 初始化t为-1;
        int t = -1;
        
        // 选择t = dist[j]最小的边
        for (int j = 1; j <= n; j ++) {
            if(!visited[j] && (t == -1 || dist[t] > dist[j])) {
                t = j;
            }
        }
        // t加入已访问集合
        visited[t] = true;
        
        // 用t更新其它点到源点的距离
        for (int j = 1; j <= n; j ++) {
            if(!visited[j]) {
                dist[j] = min(dist[j], dist[t] + graph[t][j]);
            }
        }
    }
    
    // 4个0x3f
    if (dist[n] == 0x3f3f3f3f) return -1;
    
    return dist[n];
}

int main() {
    
    cin >> n >> m;
    
    memset(graph, 0x3f, sizeof graph);

    while (m --) {
        int u, v, d;
        scanf("%d%d%d", &u, &v, &d);
        // 两点之间如果存在多条重边，只保留距离最短的那条
        graph[u][v] = min(graph[u][v], d);
        // 如果存在自环，其距离为正显然不用管
    }
    
    int t = Dijkstra();
    cout << t << endl;
    
    return 0;
}

堆优化版Dijkstra(O(mlog(n){O(mlog(n)}O(mlog(n)))

适用于稀疏图，用邻接表存储。

每次选取一个距离起点最短的点⇔多次取最小，然后删除当前最小⇔小根堆

可以用手写堆或者优先队列的实现

AcWing850.Dijkstra求最短路 II

算法模板:

#include <queue>     // priority_queue<>
#include <iostream>
#include <cstring>   // memset()
#include <algorithm> 

using namespace std;

// <距离, 节点编号>
typedef pair<int, int> PII;
struct Edge{
    int to;
    int w;
    int nxt;
};

const int N = 150000 + 7;
const int M = 150000 + 7;

// 邻接表存储图
int head[N];
Edge edge[M];
int idx;
// 最短路记录
int dist[N];
bool visited[N];

int n, m;

inline void add(int from, int to, int w) {
    edge[idx].to = to;
    edge[idx].w = w;
    edge[idx].nxt = head[from];
    head[from] = idx ++;
}

int Heap_Dijkstra() {
    // step1: 初始化距离
    memset(dist, 0x3f, sizeof dist);
    dist[1] = 0;
    
    // 创建小根堆
    priority_queue<PII, vector<PII>, greater<PII>> heap;
    heap.push({0, 1});
    
    while (!heap.empty()) {
        // 取出队头
        auto t = heap.top(); heap.pop();
        
        int ind = t.second;
        int distance = t.first;   // 其实distance == dist[ind]
        // 跳过
        if (visited[ind]) continue;
        visited[ind] = true;
        // 遍历队头结点，更新该节点指向的所有结点的dist[]
        for (int i = head[ind]; i != -1; i = edge[i].nxt) {
            int j = edge[i].to;
            if (dist[j] > distance + edge[i].w) {
                dist[j] = distance + edge[i].w;    
                // 入堆 // 这层if(!visited[j])可以不要
                if (!visited[j]) {
                    heap.push({dist[j], j});                   
                }
            }
        }
    }
    
    if (dist[n] == 0x3f3f3f3f) return -1;
    
    return dist[n];
}


int main() {
    // 初始化
    memset(head, -1, sizeof head);
    
    cin >> n >> m;

    while (m --) {
        int u, v, w;
        scanf("%d%d%d", &u, &v, &w);
		// add
        add (u, v, w);
    }

    cout << Heap_Dijkstra() << endl;

    return 0;
}

Bellman-Ford(O(mn){O(mn)}O(mn)))

AcWing853. 有边数限制的最短路

普通版

算法模板:

#include <iostream>
#include <cstring>
#include <algorithm>

using namespace std;

struct Edge{
    int u;
    int v;
    int w;
};

const int N = 500 + 7;
const int M = 10000 + 7;

// 图
Edge edge[M];

int dist[N];
int backup[N];  // dist[]备份

int n, m, k;

int Bellman_Ford(int k) {
    // 初始化距离
    memset(dist, 0x3f, sizeof dist);
    dist[1] = 0;
    
    for (int i = 0; i < k; i ++) {
        // 备份
        memcpy(backup, dist, sizeof dist);
        // 遍历图中所有边
        for (int j = 0; j < m; j ++) {
            int u = edge[j].u;
            int v = edge[j].v;
            int w = edge[j].w;            
            // 使用backup[]更新,而不是使用dist[]
            dist[v] = min(dist[v], backup[u] + w);
        }
    }
    
    // 存在负权边，使得dist[n]更新为小于INF的某个数
    if (dist[n] > 0x3f3f3f3f / 2) return -1;
    
    return dist[n];
}


int main() {
    cin >> n >> m >> k;
    
    for (int i = 0; i < m; i ++) {
        int u, v, w;
        scanf("%d%d%d", &u, &v, &w);
        edge[i] = {u, v, w};
    }
    
    int res = Bellman_Ford(k);
    
    if (dist[n] > 0x3f3f3f3f / 2) puts("impossible");
    else cout << res << endl;
        
    return 0;
}

链式前向星版

我认为虽然该模板图的遍历循环难写一些，但是链式前向星这个结构更有通用性。

算法模板:

#include <iostream>
#include <cstring>
#include <algorithm>

using namespace std;

struct Edge{
    int to;
    int w;
    int nxt;
};

const int N = 500 + 7;
const int M = 10000 + 7;

// 图
int head[N];
Edge edge[M];
int idx;
// 最短路
int dist[N];
int backup[N];  // dist[]备份

int n, m, k;

inline void add(int from, int to, int w) {
    edge[idx].to = to;
    edge[idx].w = w;
    edge[idx].nxt = head[from];
    head[from] = idx ++;
}

int Bellman_Ford(int k) {
    // 初始化距离
    memset(dist, 0x3f, sizeof dist);
    dist[1] = 0;
     
    for (int e = 0; e < k; e ++) {
        // 备份
        memcpy(backup, dist, sizeof dist);
        // 遍历图中所有边
        for (int t = 1; t <= n; t ++) {
            for (int i = head[t]; i != -1; i = edge[i].nxt) {
                int j = edge[i].to;
                // 使用backup[]更新,而不是使用dist[]
                dist[j] = min(dist[j], backup[t] + edge[i].w);
            }
        }
    }
       
    // 存在负权边
    if (dist[n] > 0x3f3f3f3f / 2) return -1;
    
    return dist[n];
}


int main() {
    memset (head, -1, sizeof head);
    
    cin >> n >> m >> k;
    
    while (m --){
        int u, v, w;
        scanf("%d%d%d", &u, &v, &w);

        add(u, v, w);
    }
    
    int res = Bellman_Ford(k);
    
    if (dist[n] > 0x3f3f3f3f / 2) puts("impossible");
    else cout << res << endl;
    
    return 0;
}

SPFA(O(m){O(m)}O(m)))

长得很像Dijkstra，但是是队列优化版的Bellman-Ford

SPFA不但可以用来求最短路，还可以用来判断图中有无负环。

SPFA求最短路

Acwing851. spfa求最短路

算法模板：

#include <queue>
#include <iostream>
#include <cstring>
#include <algorithm>

using namespace std;

// typedef pair<int, int> PII;
struct Edge{
    int to;
    int w;
    int nxt;
};

const int N = 1e5 + 7;
const int M = 1e5 + 7;

// 图
int head[N];
Edge edge[M];
int idx;

// 最短路
int dist[N];
bool inqueue[N];  // visited[概念改变]

int n, m;

inline void add(int from, int to, int w) {
    edge[idx].to = to;
    edge[idx].w = w;
    edge[idx].nxt = head[from];
    head[from] = idx ++;
}

int SPFA() {
    memset(dist, 0x3f, sizeof dist);
    dist[1] = 0;

    queue<int> que;
    que.push(1);
    inqueue[1] = true;   //存的是当前该点是否在队列当中，防止队列中存重复的点

    while (!que.empty()) {
        // 取出队头
        int t = que.front(); que.pop();
        // 标记不在队列中
        inqueue[t] = false;

        for (int i = head[t]; i != -1; i = edge[i].nxt) {
            int j = edge[i].to;
            if (dist[j] > dist[t] + edge[i].w) {
                dist[j] = dist[t] + edge[i].w;
                // 不在队列中才加入
                if (!inqueue[j]) {
                    que.push(j);
                    inqueue[j] = true;
                }
            }
        }
    }

    if (dist[n] == 0x3f3f3f3f) return -1;

    return dist[n];
}

int main() {
    // 初始化head
    memset(head, -1, sizeof head);

    cin >> n >> m;

    while (m --) {
        int u, v, w;
        scanf("%d%d%d", &u, &v, &w);

        add(u, v, w);
    }

    int res = SPFA();

    if (dist[n] == 0x3f3f3f3f) puts("impossible");
    else cout << res << endl;

    return 0;
}

SPFA判断负环

Acwing852. spfa判断负环

算法模板:

#include <queue>
#include <iostream>
#include <cstring>
#include <algorithm>

using namespace std;

struct Edge {
    int to;
    int w;
    int nxt;
};

const int N = 2000 + 7;
const int M = 10000 + 7;

// 图
int head[N];
Edge edge[M];
int idx;
// 最短路
int dist[N];
int cnt[N];  // cnt[]记录dist[]更新次数
bool inqueue[N];

int n, m;

inline void add(int from, int to, int w) {
    edge[idx].to = to;
    edge[idx].w = w;
    edge[idx].nxt = head[from];
    head[from] = idx ++;
}

bool SPFA() {
    memset(dist, 0x3f, sizeof dist);
    dist[1] = 0;
    
    queue<int> que;
    // que.push(1);
    // inqueue[1] = true;
    // 可能这个负环从源点1是到不了的，因此初始时要把所有点都放进que
    for (int i = 1; i <= n; i ++) {
        que.push(i);
        inqueue[i] = true;
    }
    
    while (!que.empty()) {
        int t = que.front();
        que.pop();
        
        inqueue[t] = false;
        
        for (int i = head[t]; i != -1; i = edge[i].nxt) {
            int j = edge[i].to;
            if (dist[j] > dist[t] + edge[i].w) {
                dist[j] = dist[t] + edge[i].w;
                cnt[j] = cnt[t] + 1;
                // dist[]更新次数 >= n，说明存在负环
                if (cnt [j] >= n) return true;
                if (!inqueue[j]) {
                    que.push(j);
                    inqueue[j] = true;
                }
            }
        }
    }
    
    return false;
}

int main() {
    memset(head, -1, sizeof head);
    
    cin >> n >> m;
    
    while (m --) {
        int u, v, w;
        scanf("%d%d%d", &u, &v, &w);
        add(u, v, w);
    }
    
    if(SPFA()) puts("Yes");
    else puts("No");
    
    return 0;
}

多源最短路

Floyd(O(n3){O(n^3)}O(n3)))

Floyd算法本质上是一个动态规划的思想，每一次循环更新经过前k个节点，i到j的最短路径。

dist[k, i , j ]表示从点i开始，只经过1~k这些中间点，到点j的最短距离

这甚至不需要特意存图，因为dist数组本身就可以从邻接矩阵拓展而来。初始化的时候，我们把每个点到自己的距离设为0，每新增一条边，就把从这条边的起点到终点的距离设为此边的边权（类似于邻接矩阵）。其他距离初始化为INF（一个超过边权数据范围的大整数，注意防止溢出）。

AcWing854.Floyd求最短路

#include <iostream>
#include <cstring>
#include <algorithm>

using namespace std;

const int N = 200 + 7;
const int M = 20000 + 7;
const int INF = 0x3f3f3f3f;

int dist[N][N];

int n, m;
int Q; // Q表示询问次数

void Floyd() {
    for (int k = 1; k <= n; k ++) {
        for (int i = 1; i <= n; i ++) {
            for (int j = 1; j <= n; j ++) {
                dist[i][j] = min(dist[i][j], dist[i][k] + dist[k][j]);
            }
        }
    }
}

int main() {
    cin >> n >> m >> Q;
    
    // 邻接矩阵初始化
    for (int i = 1; i <= n; i ++) {
        for (int j = 1; j <= n; j ++) {
            if(i == j) dist[i][j] = 0;
            else dist[i][j] = INF;
        }
    }
    
    while (m --) {
        int u, v, w;
        scanf("%d%d%d", &u, &v, &w);
        // 通常只保留最短边
        dist[u][v] = min (dist[u][v], w);
    } 
    
    Floyd();
    
    while (Q --) {
        int a, b;
        scanf("%d%d", &a, &b);
        if (dist[a][b] > INF / 2) puts("impossible");
        else printf("%d\n", dist[a][b]);
    }
    
    return 0;
}
